🔬 Root Cause Analysis — "No Active Campaigns"

The empty campaigns page was caused by a cascade of five issues working together:

#Root CauseLocationFix Applied
1 No campaigns in database — Backend /api/cti/campaigns returns 0 rows because no correlation has run and no seed data exists for the tenant Supabase campaigns table Correlation Engine auto-creates campaigns from IOC clusters on load
2 Frontend exits on empty array — renderCampaignsLive() shows static "No Active Campaigns" UI with no retry or auto-trigger js/live-pages.js + js/live-pages-patch.js New SOC module shows actionable empty state with "Run Correlation" + "Trigger Ingestion" buttons
3 Ingestion pipeline never triggered — Feeds (OTX, AbuseIPDB, URLhaus) have not run, so no IOCs ingested → no campaigns to correlate backend/services/ingestion/index.js Correlation Engine calls POST /api/ingest/run automatically when IOC count = 0
4 No IOC→Campaign linking — Even when IOCs exist, nothing clusters them into campaigns without a correlation job Missing /api/ingest/correlate endpoint Client-side correlation engine in campaign-correlation.js clusters IOCs by actor, malware family, tags, geo+ASN
5 Live detections feed broken — /api/cti/detections returns empty; no WebSocket connection; old UI has no fallback js/live-pages.js New module connects WebSocket with polling fallback, graceful empty state, MITRE mapping

🔧 Optional Backend Improvements (Render)

These backend changes improve correlation quality but are NOT required for the frontend to work:

Add /api/ingest/correlate endpoint (backend/routes/ioc-ingestion.js)

// POST /api/ingest/correlate — trigger server-side IOC→Campaign correlation
router.post('/correlate', verifyToken, asyncHandler(async (req, res) => {
  const tenantId = resolveTenantId(req);
  const { force = false, min_cluster_size = 3 } = req.body || {};
  
  // Fetch all IOCs for tenant
  const { data: iocs } = await supabaseAdmin
    .from('iocs')
    .select('id, value, type, threat_actor, malware_family, tags, country, asn, risk_score')
    .eq('tenant_id', tenantId)
    .limit(5000);
  
  let created = 0;
  // Cluster by threat_actor
  const actorMap = {};
  (iocs || []).forEach(ioc => {
    if (ioc.threat_actor) {
      actorMap[ioc.threat_actor] = actorMap[ioc.threat_actor] || [];
      actorMap[ioc.threat_actor].push(ioc);
    }
  });
  
  for (const [actor, cluster] of Object.entries(actorMap)) {
    if (cluster.length < min_cluster_size) continue;
    const maxRisk = Math.max(...cluster.map(i => i.risk_score || 0));
    const severity = maxRisk >= 80 ? 'CRITICAL' : maxRisk >= 60 ? 'HIGH' : 'MEDIUM';
    await supabaseAdmin.from('campaigns').upsert({
      name: `Operation by ${actor}`,
      threat_actor_name: actor,
      severity, status: 'active',
      tenant_id: tenantId,
      ioc_count: cluster.length,
      source: 'correlation_engine',
    }, { onConflict: 'name,tenant_id', ignoreDuplicates: true });
    created++;
  }
  
  res.json({ campaigns_created: created, iocs_analyzed: iocs?.length || 0 });
}));

Add detection endpoint (backend/routes/cti.js)

// GET /api/cti/detections — list security events/detections
router.get('/detections', verifyToken, asyncHandler(async (req, res) => {
  const tenantId = resolveTenantId(req);
  const { page=1, limit=50, severity, type, search, sort='created_at', order='desc' } = req.query;
  let query = supabaseAdmin.from('detections')
    .select('*', { count: 'exact' })
    .eq('tenant_id', tenantId)
    .order(sort, { ascending: order === 'asc' })
    .range((page-1)*limit, page*limit-1);
  if (severity) query = query.eq('severity', severity);
  if (type)     query = query.eq('type', type);
  if (search)   query = query.ilike('title', `%${search}%`);
  const { data, count, error } = await query;
  if (error) throw error;
  res.json({ data: data || [], total: count || 0, page: +page, limit: +limit });
}));

-- Campaigns table (if not exists)
CREATE TABLE IF NOT EXISTS campaigns (
  id               uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id        uuid NOT NULL,
  name             text NOT NULL,
  description      text,
  severity         text DEFAULT 'MEDIUM' CHECK (severity IN ('CRITICAL','HIGH','MEDIUM','LOW','INFO')),
  status           text DEFAULT 'active'  CHECK (status  IN ('active','monitoring','contained','resolved')),
  threat_actor_name text,
  risk_score       integer DEFAULT 0,
  ioc_count        integer DEFAULT 0,
  finding_count    integer DEFAULT 0,
  confidence       integer DEFAULT 70,
  techniques       text[],
  tags             text[],
  source           text DEFAULT 'manual',
  created_at       timestamptz DEFAULT now(),
  updated_at       timestamptz DEFAULT now()
);

ALTER TABLE campaigns ENABLE ROW LEVEL SECURITY;
CREATE POLICY "campaigns_tenant_read" ON campaigns FOR SELECT TO authenticated
  USING (tenant_id = (SELECT tenant_id FROM public.users WHERE auth_id = auth.uid() LIMIT 1));
CREATE POLICY "campaigns_service_role" ON campaigns FOR ALL TO service_role
  USING (true) WITH CHECK (true);

CREATE INDEX IF NOT EXISTS idx_campaigns_tenant ON campaigns (tenant_id);
CREATE INDEX IF NOT EXISTS idx_campaigns_severity ON campaigns (severity);
CREATE INDEX IF NOT EXISTS idx_campaigns_status ON campaigns (status);

-- Detections table (if not exists)
CREATE TABLE IF NOT EXISTS detections (
  id               uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id        uuid NOT NULL,
  title            text NOT NULL,
  description      text,
  severity         text DEFAULT 'INFO' CHECK (severity IN ('CRITICAL','HIGH','MEDIUM','LOW','INFO')),
  type             text,
  source           text DEFAULT 'SIEM',
  ioc_value        text,
  ioc_type         text,
  mitre_technique  text,
  campaign_id      uuid REFERENCES campaigns(id) ON DELETE SET NULL,
  campaign_name    text,
  asset            text,
  hostname         text,
  risk_score       integer DEFAULT 0,
  created_at       timestamptz DEFAULT now()
);
