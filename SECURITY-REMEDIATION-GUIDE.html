Backend Break-Glass Service (NEW FILE)

Emergency access lives server-side in backend/services/break-glass.js. It requires: IP allowlist, time-limited OTP (not password), explicit approval workflow, and immutable audit trail.

/**
 * backend/services/break-glass.js
 * Server-side only emergency access — never exposed to browser
 * Requires: IP allowlist + TOTP OTP + dual approval + audit log
 */
const crypto = require('crypto');
const { supabase } = require('../config/supabase');

// Only these IPs may request a break-glass session
const ALLOWED_IPS = (process.env.BREAK_GLASS_ALLOWED_IPS || '')
  .split(',').map(ip => ip.trim()).filter(Boolean);

// Break-glass OTP TTL: 10 minutes
const OTP_TTL_MS = 10 * 60 * 1000;

async function requestBreakGlass(req, adminEmail) {
  // 1. IP allowlist check
  const clientIP = req.ip || req.connection?.remoteAddress;
  if (ALLOWED_IPS.length > 0 && !ALLOWED_IPS.includes(clientIP)) {
    await auditBreakGlass('REJECTED_IP', adminEmail, clientIP);
    throw new Error('Break-glass access denied: IP not in allowlist');
  }

  // 2. Generate cryptographically random OTP (sent via out-of-band channel)
  const otp = crypto.randomInt(100000, 999999).toString();
  const otpHash = crypto.createHash('sha256').update(otp).digest('hex');
  const expiresAt = new Date(Date.now() + OTP_TTL_MS).toISOString();

  // 3. Store OTP hash (never the plaintext)
  await supabase.from('break_glass_requests').insert({
    admin_email: adminEmail, otp_hash: otpHash,
    expires_at: expiresAt, ip_address: clientIP,
    status: 'PENDING'
  });

  // 4. Audit log — every request, success or failure
  await auditBreakGlass('OTP_REQUESTED', adminEmail, clientIP);

  // 5. Send OTP via out-of-band (email/SMS/Slack) — NOT returned to caller
  await sendOTPOutOfBand(adminEmail, otp);

  return { message: 'OTP sent via secure channel. Valid for 10 minutes.' };
}

async function validateBreakGlass(req, adminEmail, otp) {
  const otpHash = crypto.createHash('sha256').update(otp).digest('hex');
  const clientIP = req.ip;

  const { data: request } = await supabase
    .from('break_glass_requests')
    .select('*')
    .eq('admin_email', adminEmail)
    .eq('otp_hash', otpHash)
    .eq('status', 'PENDING')
    .gt('expires_at', new Date().toISOString())
    .single();

  if (!request) {
    await auditBreakGlass('INVALID_OTP', adminEmail, clientIP);
    throw new Error('Invalid or expired OTP');
  }

  // Mark used (single-use)
  await supabase.from('break_glass_requests')
    .update({ status: 'USED', used_at: new Date().toISOString() })
    .eq('id', request.id);

  // Issue time-limited (1 hour) session with full audit trail
  const sessionToken = crypto.randomBytes(48).toString('hex');
  await supabase.from('break_glass_sessions').insert({
    admin_email: adminEmail, session_token_hash: crypto.createHash('sha256').update(sessionToken).digest('hex'),
    ip_address: clientIP, expires_at: new Date(Date.now() + 3600000).toISOString()
  });

  await auditBreakGlass('SESSION_GRANTED', adminEmail, clientIP);
  return { session_token: sessionToken, expires_in: 3600 };
}

async function auditBreakGlass(event, email, ip) {
  await supabase.from('break_glass_audit').insert({
    event, email, ip_address: ip, timestamp: new Date().toISOString()
  });
}

module.exports = { requestBreakGlass, validateBreakGlass };

F2 · Bearer Tokens & Profiles in localStorage
Critical   Files: js/auth-persistent.js, js/auth-interceptor.js
🔴

XSS-Readable Token Storage — OWASP A07:2021

Every JS payload that achieves XSS on this origin can call localStorage.getItem('wadjet_access_token') and get a bearer token valid for 15 minutes — or the refresh token valid for 30 days. The full user profile (role, permissions, tenant_id) is also stored in plaintext under wadjet_user_profile. For an admin on a shared workstation, the token persists after the session ends.

Secure Design: httpOnly Cookie + BFF Pattern

── BEFORE (Insecure) ──────────────────────────────────────────────────────── Browser JS → POST /api/auth/login → Backend returns { token, refresh_token } ↓ JS stores in localStorage (XSS-readable) ↓ localStorage['wadjet_access_token'] = 'eyJhbGci...' ← readable by ANY JS on origin localStorage['wadjet_refresh_token'] = 'abc123...' ← 30-day persistent exposure localStorage['wadjet_user_profile'] = '{role:super_admin,...}' ── AFTER (Secure — httpOnly Cookies) ──────────────────────────────────────── Browser JS → POST /api/auth/login → Backend sets httpOnly cookies: Set-Cookie: waj_session=<token>; HttpOnly; Secure; SameSite=Strict; Path=/api Set-Cookie: waj_refresh=<refresh>; HttpOnly; Secure; SameSite=Strict; Path=/api/auth/refresh Returns ONLY: { user: { name, role, tenant_name } } ← no tokens in body ↓ Cookies sent automatically by browser on same-origin requests ↓ JS cannot read waj_session — XSS cannot steal it Tab close clears session cookie (not refresh — refresh is path-scoped)

Backend: Cookie-Reading Auth Middleware (backend/middleware/auth.js — patch)

// backend/middleware/auth.js — extractToken with cookie support
function extractToken(req) {
  // 1. httpOnly cookie (preferred — XSS-safe)
  if (req.cookies?.waj_session) {
    return { token: req.cookies.waj_session, source: 'httpOnly-cookie' };
  }
  // 2. Authorization header (for API clients, mobile apps, Postman)
  const authHeader = req.headers['authorization'];
  if (authHeader?.startsWith('Bearer ')) {
    return { token: authHeader.slice(7), source: 'bearer-header' };
  }
  // 3. X-Access-Token (legacy — deprecate in v7.0)
  if (req.headers['x-access-token']) {
    return { token: req.headers['x-access-token'], source: 'x-access-token' };
  }
  return { token: null, source: null };
}

/**
 * js/auth-secure.js — Replaces auth-persistent.js + auth-interceptor.js
 * Secure session model: NO tokens in localStorage/sessionStorage
 * Tokens live in httpOnly cookies (server-managed)
 * Frontend only stores: non-sensitive display data (name, role label, avatar)
 */
(function(global) {
  const DISPLAY_KEY = 'waj_display'; // NON-sensitive only

  // ── What we store client-side (no token, no permissions object) ──
  const SecureSession = {
    save(user) {
      // ONLY store display-safe fields — role is a label, not an auth claim
      sessionStorage.setItem(DISPLAY_KEY, JSON.stringify({
        name:        user.name,
        email:       user.email,
        role_label:  user.role,        // UI display only
        tenant_name: user.tenant_name,
        avatar:      user.avatar,
        session_id:  user.session_id,  // for display/revocation UI only
      }));
      // Intentionally NOT stored: token, refresh_token, permissions array
    },
    get() {
      try { return JSON.parse(sessionStorage.getItem(DISPLAY_KEY) || 'null'); }
      catch { return null; }
    },
    clear() {
      sessionStorage.removeItem(DISPLAY_KEY);
      // Also clear any legacy localStorage keys from old versions
      ['wadjet_access_token','wadjet_refresh_token','wadjet_user_profile',
       'we_access_token','we_refresh_token','we_user',
       'tp_access_token','wadjet_token_expires_at']
        .forEach(k => { localStorage.removeItem(k); sessionStorage.removeItem(k); });
    },
    exists() { return !!this.get(); }
  };

  // ── authFetch — cookies sent automatically, no manual token injection ──
  async function authFetch(pathOrUrl, opts = {}) {
    const url = pathOrUrl.startsWith('http') ? pathOrUrl
      : `${API_BASE()}/api${pathOrUrl}`;

    const response = await fetch(url, {
      ...opts,
      credentials: 'include', // ← sends httpOnly cookies automatically
      headers: { 'Content-Type': 'application/json', ...(opts.headers || {}) },
    });

    if (response.status === 401) {
      // Try silent refresh (cookie-based)
      const refreshed = await _silentRefresh();
      if (refreshed) {
        return fetch(url, { ...opts, credentials: 'include',
          headers: { 'Content-Type': 'application/json', ...(opts.headers||{}) }
        });
      }
      SecureSession.clear();
      showSessionExpiredBanner();
      throw new Error('Session expired');
    }
    if (!response.ok) throw new Error(`HTTP ${response.status}`);
    return response.json();
  }

  async function _silentRefresh() {
    try {
      // Cookie is sent automatically — no token in body needed
      const r = await fetch(`${API_BASE()}/api/auth/refresh`, {
        method: 'POST', credentials: 'include',
        headers: { 'Content-Type': 'application/json' }
      });
      return r.ok;
    } catch { return false; }
  }

  global.SecureSession = SecureSession;
  global.authFetch     = authFetch;
})(window);

    return {
      valid: false,
      errors: [`Type mismatch: value looks like '${detected.type}' but declared as '${ioc.type}'`],
      detected
    };
  }
  return { valid: true, canonical: detected.canonical || ioc.value, detected };
}

function canonicalizeIOC(ioc) {
  const result = validateIOC(ioc);
  if (!result.valid) throw new Error(`IOC validation failed: ${result.errors.join(', ')}`);
  return {
    ...ioc,
    value:        result.canonical,
    type:         result.detected?.type || ioc.type,
    type_subtype: result.detected?.subtype || null,
    normalized_at: new Date().toISOString(),
  };
}

module.exports = { detectType, validateIOC, canonicalizeIOC };

  const valid = [], invalid = [];

      valid.push(canonicalizeIOC(raw));

      invalid.map(i => ({ feed: feedName, raw_value: i.raw.value, error: i.error,
        raw_type: i.raw.type, occurred_at: new Date().toISOString() }))
    );
  }

}
